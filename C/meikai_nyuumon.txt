#+BEGIN_SRC C
  #include <stdio.h>

  void rev_string(char s[][128], int n)
  {
      char tmp[n][128];
      int i, j, length;
      for (i = 0; i < n; i++) {
          length = 0;
          for (j = 0; s[i][j] != '\0'; j++) {
              tmp[i][j] = s[i][j];
              length++;
          }
          for (j = 0; j < length; j++) {
            s[i][length - j - 1] = tmp[i][j];
          }
      }
  }
